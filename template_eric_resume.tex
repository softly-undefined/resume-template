%--------------------New commands and renewed commands--------------------

%Changes the \maketitle command to place website name(s) on the top left of the page, 
%email and phone number top middle, and address top right
\renewcommand{\maketitle}[6]{
    \begin{center}
        {\LARGE\bfseries\theauthor}
    \end{center}

    \vspace{-20pt}

    \begin{multicols}{3}
        \begin{flushleft}
            #1 \\
            #2
        \end{flushleft}

        \begin{center}
            #3 \\
            #4
        \end{center}

        \begin{flushright}
            #5 \\
            #6
        \end{flushright}
    \end{multicols}
}

%Creates a right-aligned date range in italics
\renewcommand{\date}[1]{
    \hfill{\normalsize\textit{#1}}
}

%Establishes the name of the award with a small gap after the company name
\newcommand{\awardname}[1]{
    \hspace{0em}
    {#1}
}

\newcommand{\relevantcoursework}{
    {\bfseries Relevant Coursework: }
}

%makes the \titlerule command, which makes horizontal lines, make a slightly
%better looking and spaced out horizontal line
\renewcommand{\titlerule}{
    \vspace*{0.5ex}
    \noindent\hrulefill\par
}

\setlist[itemize]{itemsep=-5pt, topsep=0pt} %Changes spacing of itemize calls
\pagestyle{empty} %removes page numbers
